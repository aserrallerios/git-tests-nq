CREATE TABLE `personality` (
	`id` int(11) NOT NULL AUTO_INCREMENT,
	`project` varchar(128) NOT NULL,
	`code` int(11) NOT NULL,
	`evaluations` int(11) DEFAULT 0,
	`spontaneous_ack` int(11) DEFAULT 0,
	`suggested_ack` int(11) DEFAULT 0,
	UNIQUE KEY `project_code` (`project`, `code`),
	KEY `evaluations` (`project`,`evaluations`),
	KEY `spontaneous_ack` (`project`,`spontaneous_ack`),
	KEY `suggested_ack` (`project`,`suggested_ack`),
	PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
